# * là lấy tất cả column trong table đó
SELECT * FROM users
WHERE age BETWEEN 30 and 35 -- điều kiện lấy dữ liệu
# tăng dần : asc
# giảm dần: desc
ORDER BY age desc

# tìm tất cả những người có chữ d in ra
SELECT * FROM users
WHERE full_name like '%D%'
# %D tìm những người có chữ d cuối cùng trong tên
# D% tìm những người có chữ d đầu tiên tron tên

# kết hợp nhiều điều kiện
